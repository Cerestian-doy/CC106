<script>
async function fetchGitHubFile(){

    const output = document.getElementById("file-content");
    output.textContent="> Establishing secure connection...\n";

    const owner = "Cerestian-doy";
    const repo = "CC106";
    const filePath = "sampletextfile.txt";
    const branch = "main";

    const url = `https://api.github.com/repos/${owner}/${repo}/contents/${filePath}?ref=${branch}`;

    try{
        const response = await fetch(url);

        if(!response.ok){
            throw new Error(`ERROR ${response.status}`);
        }

        output.textContent += "> Access granted...\n> Decoding file...\n";

        const data = await response.json();
        const content = atob(data.content);

        setTimeout(()=>{
            output.textContent = content;
        },800);

    }catch(error){
        output.textContent="‚ùå SYSTEM FAILURE\n"+error;
    }
}
